4.打印九九乘法表

            【解法】

            for (var i = 1; i <= 9; i++) {
                for (var j = 1; j <= i; j++) {
                    result += j + "x" + i + "=" + (j * i) + "\t";
                }
                result += "\n";
            }
            console.log(result);

5.翻转数组

            将var arr = [5, 4, 3, 2, 1];翻转

            【解法】

            function reverse(arr) {
                var newArr = [];
                for (var i = arr.length - 1; i >= 0; i--) {
                    newArr[newArr.length] = arr[i];
                }
                return newArr;
            }
            var arr = [5, 4, 3, 2, 1];
            console.log(reverse(arr));

6.数组排序（冒泡排序）

            将var arr1 = [3, 1, 6, 4, 2];从小到大排序

            【解法】

            function sort(arr) {
                for (var i = 0; i < arr.length - 1; i++) {
                    for (var j = 0; j <= arr.length - i - 1; j++) {
                        if (arr[j] > arr[j + 1]) {
                            var temp = arr[j];
                            arr[j] = arr[j + 1];
                            arr[j + 1] = temp;
                        }
                    }
                }
                return arr;
            }
            var arr1 = [3, 1, 6, 4, 2];
            console.log(sort(arr1));

7.判断一个数是否为质数

            // 判断传入的一个数值是否是质数
            // 质数：从2开始只能被1和自身整除的数
            function isPrimeNumber(a) {
                for (var i = 2; i < a; i++) {
                    if (a % i === 0) {
                        return false;
                    }
                }
                return true;
            }
            console.log(isPrimeNumber(15)); // 15不是质数

8.求['red','blue','green','pink','red']中red出现的位置及次数

            var str = ['red', 'blue', 'green', 'pink', 'red'];
            var index = str.indexOf("red"); // 出现的位置
            var num = 0; // 出现的次数
            // console.log(index);
            while (index != -1) { // 找到了red
                num++;
                console.log('出现的位置：' + index);
                // 往后查找
                index = str.indexOf("red", index + 1);
            }
            console.log('出现的次数：' + num);

10.字符串操作

            var str = "abaasdffggghhjjkkgfddsssss3444343";

            (1)将所有的a替换成-

            console.log(str.replace(/a/g, "-"));


            (2)统计字符串中出现次数最多的字符和出现的次数

            function count(str) {
                var o = {};
                // 遍历数组，统计每个字符出现的次数
                for (var i = 0; i < str.length; i++) {
                    var char = str.charAt(i);
                    if (o[char]) { // 对象中存在该字符
                        o[char]++;
                    } else { // 对象中没有改
                        o[char] = 1;
                    }
                }
                // 遍历对象，统计出现次数最多的字符
                var max = 0;
                var ch = "";
                for (var k in o) {
                    // k:属性名 o[k]:属性值
                    if (o[k] > max) {
                        max = o[k];
                        ch = k;
                    }
                }
                return ch + ":" + o[ch];
            }
            console.log(count(str));
